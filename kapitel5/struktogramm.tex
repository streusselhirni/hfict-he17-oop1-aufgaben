\begin{struktogramm}(100,100)
    \sProofOn
    \assign{Zahl eingeben}
    \ifthenelse{3}{4}{Zahl gerade?}{Ja}{Nein}
        \assign{Ausgabe "Die Zahl ist gerade"}
        \change
        \assign{Ausgabe "Die Zahl ist ungerade}
        \ifthenelse{1}{1}{Zahl kleiner als 0?}{Ja}{Nein}
            \assign{Ausgabe "Die Zahl ist kleiner 0"}
            \change
            \assign{Ausgabe "Die Zahl ist größer/gleich 0}
        \ifend
    \ifend
\end{struktogramm}

\begin{struktogramm}(100,50)

        \assign{Ausgabe "Die Zahl ist gerade"}

        \switch{Nein}
            \assign{Ausgabe "Die Zahl ist größer/gleich 0}
        \caseend
    \caseend
\end{struktogramm}

\section{Aufgabe 5.2}
\subsection{Mit For Schleife}
\begin{struktogramm}(100,50)
    \assign{Basis $b$ eingeben}
    \assign{Exponent $e$ eingeben}
    \assign{$x$ mit Wert von $b$ initialisieren}
    \while{int i = 1; i < e; ++i}
        \assign{$x = x \times b$}
    \whileend
    \assign{$b^e = x$}
\end{struktogramm}

\subsection{Mit While Schleife}
\begin{struktogramm}(100,50)
    \assign{Basis $b$ eingeben}
    \assign{Exponent $e$ eingeben}
    \assign{$x$ mit Wert von $b$ initialisieren}
    \assign{Zählvariable $x = 1$ initialisieren.}
    \while{Solange $i < e$}
        \assign{$x = x \times b$}
        \assign{$++i$}
    \whileend
\assign{$b^e = x$}
\end{struktogramm}

\subsection{Mit DoWhile Schleife}
\begin{struktogramm}(100,50)
    \assign{Basis $b$ eingeben}
    \assign{Exponent $e$ eingeben}
    \assign{$x$ mit Wert von $b$ initialisieren}
    \until{int i = 1; i < e; ++i}
        \assign{$x = x \times b$}
    \untilend
    \assign{$b^e = x$}
\end{struktogramm}
